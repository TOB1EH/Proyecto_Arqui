// ---> CONTRASEÑA <---

0        SUB R0, R15, R15            // SETEA RO EN CERO
1        LDR R9, [R0, #0x4]          // GUARDA LA CONTRASEÑA DE DMEM EN R9
2        ADD R5, R0, #3              // CONTADOR DE 3 INTENTOS EN R5
3        ADD R7, R0, #1              // SETEA R7 EN UNO PARA DUPLICAR
4        ADD R6, R0, #1              // SETEA Y USO R6 PARA ENCENDER LEDS
5        STR R6, [R0, #0x800]        // ENCIENDO LEDS CARGADOS EN R6
6        LDR R11, [R0, #0x800]       // LECTURA DEL PULSADOR
7        ANDS R11, R11, #0x80        // COMPARA SI SE PRESIONO EL BOTON
8        BNE #24                     // VUELVE A LECTURA DEL PULSADOR (SI SE PULSA, APAGO BNE)
9        LDR R12, [R0, #0x8]         // SETEA R5 CON DELAY ANTIRREBOTE PULSADOR
10       SUBS R12, R12, #1           // RESTA DELAY (EMPIEZA DELAY)
11       BNE #40                     // VUELVE AL SUBS PARA RESTAR TIEMPO AL DELAY (CUANDO TERMINA, APAGA BNE)
12       LDR R12, [R0, #0x800]       // LEE LLAVES Y SETEA EN R12
13       AND R12, R12, R9            // COMPARO LAS LLAVES CON EL REGISTRO R9
14       SUBS R12, R12, R9           // SI SON IGUALES R12 ES IGUAL A CERO (CONTRASEÑA CORRECTA)
15       BNE #68                     // SI NO SON IGUALES, SALTO AL DISMINUCION DE INTENTOS
16       B #88                       // SI SON IGUALES SALTO AL MENU
                                     // DISMINUCION DE INTENTOS:
17       ADD R7, R7, R7              // DUPLICA R7 (PRIMERA ITERACION R7 = 1)
18       ADD R6, R6, R7              // SUMA R7 A R6 PARA PRENDER EL SEGUNDO LED DE INDICADOR DE INTENTOS
19       SUBS R5, R5, #1             // DECREMENTA EN 1 LA CANTIDAD DE INTENTOS
20       BNE #20                     // VUELVE ENCIENDO LEDS INDICADOR DE INTENTOS, (APAGA BNE CUANDO SE ACABAN LOS INTENTOS)
21       B #100                      // SE ACABAN LOS INTENTOS, SALTO A APAGAR SISTEMA
                                     // MENU:
22       ADD R1, R0, #0xff           // SETEA R1 PARA ENCENDER TODOS LOS LEDS
23       STR R1, [R0, #0x800]        // ENCIENDE TODOS LOS LEDS USANDO R1
24       B #88                       // REPITE EL BUCLE MENU INFINITAMENTE
                                     // APAGAR SISTEMA:
25       ADD R1, R0, R0              // SETEA R1 EN CERO
26       STR R1, [R0, #0x800]        // NO ENCIENDO NINGUN LED
27       B #100                      // REPITE INFINITAMENTE EL BUCLE PARA ESPERAR RESET
